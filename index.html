      if(/rph|rpt|dskp|modul|borang|dokumen|template|analisis|pemantauan/.test(v)) return "Dokumen & Borang Rasmi";

      if(/takwim|jadual|waktu|akademik|kelas|bilik/.test(v)) return "Takwim & Jadual";

      if(/carta|jawatankuasa|struktur|organisasi|alir/.test(v)) return "Carta & Jawatankuasa";

      if(/kursus|hebah|pengumuman|notifikasi|bengkel/.test(v)) return "Makluman & Hebahan Dalaman";

      return "Dokumen & Borang Rasmi";

    function autoDescription(category, name=""){

      const subject = name.trim() ? name.trim() : "Item ini";

      const map = {

        "Aplikasi Pengurusan Guru": `${subject} membantu guru mengurus kehadiran, makluman dan urusan lantikan secara digital.`,

        "Aplikasi Pentaksiran & Murid": `${subject} digunakan untuk pentaksiran, profil murid dan pemantauan akademik secara digital.`,

        "Aplikasi PdP & Aktiviti Harian": `${subject} merekod aktiviti PdP, bacaan pagi atau aktiviti harian murid secara sistematik.`,

        "Dokumen & Borang Rasmi": `${subject} ialah dokumen rujukan atau borang rasmi kurikulum yang boleh dibuka dan dimuat turun.`,

        "Takwim & Jadual": `${subject} memaparkan takwim atau jadual terkini untuk pengurusan masa dan aktiviti sekolah.`,

        "Carta & Jawatankuasa": `${subject} menunjukkan struktur jawatankuasa, carta organisasi atau carta alir proses kerja sekolah.`,

        "Makluman & Hebahan Dalaman": `${subject} memaparkan makluman terkini berkaitan aktiviti, kursus atau hebahan kurikulum.`

      };

      return map[category] || `${subject} telah dimasukkan ke Portal Pengurusan Kurikulum Digital.`;

    }


    function suggestIcon(name, category){

      const v = normalize(name);

      if(/bacaan/.test(v)) return "📖";

      if(/mesyuarat|kehadiran/.test(v)) return "📋";

      if(/makluman|hebah/.test(v)) return "✉️";

      if(/pbd|pentaksiran/.test(v)) return "🧾";

      if(/delima|id/.test(v)) return "🪪";

      if(/surat|lantikan/.test(v)) return "📑";

      if(/jadual|takwim/.test(v)) return "🗓️";

      if(/carta|organisasi|struktur/.test(v)) return "🔗";

      if(/rph|rpt|dskp|modul/.test(v)) return "📚";

      if(/borang|dokumen/.test(v)) return "📄";

      return category.includes("Aplikasi") ? "🖥️" : "📌";

    }
